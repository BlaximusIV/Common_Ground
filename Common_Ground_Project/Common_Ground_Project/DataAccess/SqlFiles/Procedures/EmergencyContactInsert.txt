USE MASTER
GO

CREATE PROCEDURE dbo.EmergencyContactInsert
(
	@ParticipantID INT, 
	@Name VARCHAR(50), 
	@Phone VARCHAR(50), 
	@Email VARCHAR(50), 
	@NewID INT OUTPUT
)
AS 
	SET NOCOUNT ON;

	INSERT INTO Master.dbo.EmergencyContact
	(
		ParticipantID, 
		Name, 
		Phone, 
		Email
	)
	VALUES 
	(
		@ParticipantID, 
		@Name, 
		@Phone, 
		@Email
	)

	@NewID = SCOPE_IDENTITY()

GO
