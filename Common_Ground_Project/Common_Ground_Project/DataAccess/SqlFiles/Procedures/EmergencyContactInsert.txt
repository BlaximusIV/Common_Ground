USE MASTER
GO

CREATE PROCEDURE dbo.EmergencyContactInsert
(
	@ParticipantID, 
	@Name, 
	@Phone, 
	@Email, 
	@NewID INT OUTPUT
)
AS 
	SET NOCOUNT ON;

	INSERT INTO Master.dbo.EmergencyContact
	(
		EmergencyContactID, 
		ParticipantID, 
		Name, 
		Phone, 
		Email
	)
	VALUES 
	(
		EmergencyContactID, 
		ParticipantID, 
		Name, 
		Phone, 
		Email
	)

	@NewID = SCOPE_IDENTITY()

GO
