CREATE TABLE Individual(
	IndividualID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
	IndividualTypeID INT NOT NULL,
	LastName VARCHAR(50) NOT NULL,
	FirstName VARCHAR(50) NOT NULL,
	PhoneNumber VARCHAR(50) NOT NULL,
	Email VARCHAR(50) NOT NULL,
	StreetAddress VARCHAR(50) NOT NULL,
	City VARCHAR(50) NOT NULL,
	[State] VARCHAR(2) NOT NULL,
	ZipCode INT NOT NULL,
	BirthDay DATE NOT NULL,
	IsFrequentCaller BIT NOT NULL DEFAULT(0),
	IsWaiverSigned BIT NOT NULL DEFAULT(0),
	IsMediaReleased BIT NOT NULL DEFAULT(0)
)
CREATE TABLE IndividualActivity(
	IndividualID INT NOT NULL,
	ActivityID INT NOT NULL
)

CREATE TABLE EmergencyContact(
	EmergencyContactID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
	IndividualID INT NOT NULL,
	[Name] VARCHAR(50) NOT NULL,
	Phone VARCHAR(15) NOT NULL,
	Email VARCHAR(50) NOT NULL,
)

CREATE TABLE IndividualNote(
	IndividualNoteID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
	IndividualID INT NOT NULL,
	Note VARCHAR(50) NOT NULL,
	[DateTime] DATETIME NOT NULL
)

CREATE TABLE IndividualType(
	IndividualTypeID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
	[Name] VARCHAR(50) NOT NULL
)

CREATE TABLE Activity(
	ActivityID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
	Title VARCHAR(50) NOT NULL,
	TripLeader INT NOT NULL,
	[Description] VARCHAR(250) NOT NULL,
	[Location] VARCHAR(50),
	StartDate DATETIME NOT NULL,
	EndDate DATETIME NOT NULL,
	PickUpTime TIME NOT NULL,
	DropOffTime TIME NOT NULL,
	StaffArrivalTime TIME NOT NULL,
	Cost DECIMAL NOT NULL
)

CREATE TABLE ActivityVehicle(
	ActivityID INT NOT NULL,
	VehicleID INT NOT NULL
)

CREATE TABLE ActivityNote(
	ActivityNoteID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
	ActivityID INT NOT NULL,
	Note VARCHAR(MAX) NOT NULL,
	[DateTime] DATETIME NOT NULL
)

CREATE TABLE Vehicle(
	VehicleID INT IDENTITY(1,1) PRIMARY KEY NOT NULL,
	[Name] VARCHAR(50) NOT NULL,
	[Description] VARCHAR(250) NOT NULL,
	VIN VARCHAR(50) NOT NULL
)
